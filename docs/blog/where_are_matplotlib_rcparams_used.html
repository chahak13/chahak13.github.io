Matplotlib has a matplotlibrc file that sets up many different values that in one way or other affect the usage of the library. This note tries to document which rcParams value is used where in matplotlib. While the comments are useful, they don’t always list the secondary effects of setting some values (ex. linewidth). This should hopefully address that too.

import os

def generate_table(lines):
    params = {}
    for line in lines:
        if line.startswith("## * "):
            # print(f"\n{line[3:-2].strip()}")
            topic = line[4:-2].strip()
            params[topic] = []
        elif not line[1:].strip().startswith("#") and line[1:] != "":
            tup = line[1:].split(":")
            if len(tup) > 1:
                # print(f"| {tup[0]} | |")
                params[topic].append(tup[0])
    return params

import ast

def find_function_name(filepath, lineno):
    with open(filepath, 'r') as f:
        lines = f.read()
        root = ast.parse(lines)
        functions = [n for n in root.body if isinstance(n, ast.FunctionDef)]
        for fn in functions:
            if fn.lineno <= lineno <= fn.end_lineno:
                return fn.name
        classes = [n for n in root.body if isinstance(n, ast.ClassDef)]
        for cls in classes:
            methods = [n for n in cls.body if isinstance(n, ast.FunctionDef)]
            for method in methods:
                if method.lineno <= lineno <= method.end_lineno:
                    return f"{cls.name}.{method.name}"
    return None
